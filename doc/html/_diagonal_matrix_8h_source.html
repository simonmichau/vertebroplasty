390template<typename Derived>
391bool MatrixBase<Derived>::isDiagonal(const RealScalar& prec) const
392{
393 if(cols() != rows()) return false;
394 RealScalar maxAbsOnDiagonal = static_cast<RealScalar>(-1);
395 for(Index j = 0; j < cols(); ++j)
396 {
397 RealScalar absOnDiagonal = numext::abs(coeff(j,j));
398 if(absOnDiagonal > maxAbsOnDiagonal) maxAbsOnDiagonal = absOnDiagonal;
399 }
400 for(Index j = 0; j < cols(); ++j)
401 for(Index i = 0; i < j; ++i)
402 {
403 if(!internal::isMuchSmallerThan(coeff(i, j), maxAbsOnDiagonal, prec)) return false;
404 if(!internal::isMuchSmallerThan(coeff(j, i), maxAbsOnDiagonal, prec)) return false;
405 }
406 return true;
407}

417// Diagonal matrix to Dense assignment
418template< typename DstXprType, typename SrcXprType, typename Functor>
419struct Assignment<DstXprType, SrcXprType, Functor, Diagonal2Dense>
420{
421 static void run(DstXprType &dst, const SrcXprType &src, const internal::assign_op<typename DstXprType::Scalar,typename SrcXprType::Scalar> &/*func*/)
422 {
423 Index dstRows = src.rows();
424 Index dstCols = src.cols();
425 if((dst.rows()!=dstRows) || (dst.cols()!=dstCols))
426 dst.resize(dstRows, dstCols);
427
428 dst.setZero();
429 dst.diagonal() = src.diagonal();
430 }
431
432 static void run(DstXprType &dst, const SrcXprType &src, const internal::add_assign_op<typename DstXprType::Scalar,typename SrcXprType::Scalar> &/*func*/)
433 { dst.diagonal() += src.diagonal(); }
434
435 static void run(DstXprType &dst, const SrcXprType &src, const internal::sub_assign_op<typename DstXprType::Scalar,typename SrcXprType::Scalar> &/*func*/)
436 { dst.diagonal() -= src.diagonal(); }
