112template<typename Scalar>
113struct functor_traits<scalar_conjugate_op<Scalar> >
114{
115 enum {
116 Cost = 0,
117 // Yes the cost is zero even for complexes because in most cases for which
118 // the cost is used, conjugation turns to be a no-op. Some examples:
119 // cost(a*conj(b)) == cost(a*b)
120 // cost(a+conj(b)) == cost(a+b)
121 // <etc.
122 // If we don't set it to zero, then:
123 // A.conjugate().lazyProduct(B.conjugate())
124 // will bake its operands. We definitely don't want that!
125 PacketAccess = packet_traits<Scalar>::HasConj
126 };
127};

269template <typename Scalar>
270struct functor_traits<scalar_exp_op<Scalar> > {
271 enum {
272 PacketAccess = packet_traits<Scalar>::HasExp,
273 // The following numbers are based on the AVX implementation.
274#ifdef EIGEN_VECTORIZE_FMA
275 // Haswell can issue 2 add/mul/madd per cycle.
276 Cost =
277 (sizeof(Scalar) == 4
278 // float: 8 pmadd, 4 pmul, 2 padd/psub, 6 other
279 ? (8 * NumTraits<Scalar>::AddCost + 6 * NumTraits<Scalar>::MulCost)
280 // double: 7 pmadd, 5 pmul, 3 padd/psub, 1 div, 13 other
281 : (14 * NumTraits<Scalar>::AddCost +
282 6 * NumTraits<Scalar>::MulCost +
283 scalar_div_cost<Scalar,packet_traits<Scalar>::HasDiv>::value))
284#else
285 Cost =
286 (sizeof(Scalar) == 4
287 // float: 7 pmadd, 6 pmul, 4 padd/psub, 10 other
288 ? (21 * NumTraits<Scalar>::AddCost + 13 * NumTraits<Scalar>::MulCost)
289 // double: 7 pmadd, 5 pmul, 3 padd/psub, 1 div, 13 other
290 : (23 * NumTraits<Scalar>::AddCost +
291 12 * NumTraits<Scalar>::MulCost +
292 scalar_div_cost<Scalar,packet_traits<Scalar>::HasDiv>::value))
293#endif
294 };
295};

1007template <>
1008struct scalar_logistic_op<float> {
1009 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE float operator()(const float& x) const {
1010 // Truncate at the first point where the interpolant is exactly one.
1011 const float cst_exp_hi = 16.6355324f;
1012 const float e = numext::exp(numext::mini(x, cst_exp_hi));
1013 return e / (1.0f + e);
1014 }
1015
1016 template <typename Packet>
1017 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE Packet
1018 packetOp(const Packet& _x) const {
1019 const Packet cst_zero = pset1<Packet>(0.0f);
1020 const Packet cst_one = pset1<Packet>(1.0f);
1021 const Packet cst_half = pset1<Packet>(0.5f);
1022 // Truncate at the first point where the interpolant is exactly one.
1023 const Packet cst_exp_hi = pset1<Packet>(16.6355324f);
1024 const Packet cst_exp_lo = pset1<Packet>(-104.f);
1025
1026 // Clamp x to the non-trivial range where S(x). Outside this
1027 // interval the correctly rounded value of S(x) is either zero
1028 // or one.
1029 Packet zero_mask = pcmp_lt(_x, cst_exp_lo);
1030 Packet x = pmin(_x, cst_exp_hi);
1031
1032 // 1. Multiplicative range reduction:
1033 // Reduce the range of x by a factor of 2. This avoids having
1034 // to compute exp(x) accurately where the result is a denormalized
1035 // value.
1036 x = pmul(x, cst_half);
1037
1038 // 2. Subtractive range reduction:
1039 // Express exp(x) as exp(m*ln(2) + r) = 2^m*exp(r), start by extracting
1040 // m = floor(x/ln(2) + 0.5), such that x = m*ln(2) + r.
1041 const Packet cst_cephes_LOG2EF = pset1<Packet>(1.44269504088896341f);
1042 Packet m = pfloor(pmadd(x, cst_cephes_LOG2EF, cst_half));
1043 // Get r = x - m*ln(2). We use a trick from Cephes where the term
1044 // m*ln(2) is subtracted out in two parts, m*C1+m*C2 = m*ln(2),
1045 // to avoid accumulating truncation errors.
1046 const Packet cst_cephes_exp_C1 = pset1<Packet>(-0.693359375f);
1047 const Packet cst_cephes_exp_C2 = pset1<Packet>(2.12194440e-4f);
1048 Packet r = pmadd(m, cst_cephes_exp_C1, x);
1049 r = pmadd(m, cst_cephes_exp_C2, r);
1050
1051 // 3. Compute an approximation to exp(r) using a degree 5 minimax polynomial.
1052 // We compute even and odd terms separately to increase instruction level
1053 // parallelism.
1054 Packet r2 = pmul(r, r);
1055 const Packet cst_p2 = pset1<Packet>(0.49999141693115234375f);
1056 const Packet cst_p3 = pset1<Packet>(0.16666877269744873046875f);
1057 const Packet cst_p4 = pset1<Packet>(4.1898667812347412109375e-2f);
1058 const Packet cst_p5 = pset1<Packet>(8.33471305668354034423828125e-3f);
1059
1060 const Packet p_even = pmadd(r2, cst_p4, cst_p2);
1061 const Packet p_odd = pmadd(r2, cst_p5, cst_p3);
1062 const Packet p_low = padd(r, cst_one);
1063 Packet p = pmadd(r, p_odd, p_even);
1064 p = pmadd(r2, p, p_low);
1065
1066 // 4. Undo subtractive range reduction exp(m*ln(2) + r) = 2^m * exp(r).
1067 Packet e = pldexp_fast_impl<Packet>::run(p, m);
1068
1069 // 5. Undo multiplicative range reduction by using exp(r) = exp(r/2)^2.
1070 e = pmul(e, e);
1071
1072 // Return exp(x) / (1 + exp(x))
1073 return pselect(zero_mask, cst_zero, pdiv(e, padd(cst_one, e)));
1074 }
1075};
