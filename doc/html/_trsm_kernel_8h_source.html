99template <typename Scalar>
100int64_t avx512_trsm_cutoff(int64_t L2Size, int64_t N, double L2Cap) {
101 const int64_t U3 = 3 * packet_traits<Scalar>::size;
102 const int64_t MaxNb = 5 * U3;
103 int64_t Nb = std::min(MaxNb, N);
104 double cutoff_d =
105 (((L2Size * L2Cap) / (sizeof(Scalar))) - (EIGEN_AVX_MAX_NUM_ROW)*Nb) / ((EIGEN_AVX_MAX_NUM_ROW) + Nb);
106 int64_t cutoff_l = static_cast<int64_t>(cutoff_d);
107 return (cutoff_l / EIGEN_AVX_MAX_NUM_ROW) * EIGEN_AVX_MAX_NUM_ROW;
108}

114template <typename Scalar, typename vec, int64_t unrollM, int64_t unrollN, bool remM, bool remN>
115static EIGEN_ALWAYS_INLINE void transStoreC(PacketBlock<vec, EIGEN_ARCH_DEFAULT_NUMBER_OF_REGISTERS> &zmm,
116 Scalar *C_arr, int64_t LDC, int64_t remM_ = 0, int64_t remN_ = 0) {
117 EIGEN_UNUSED_VARIABLE(remN_);
118 EIGEN_UNUSED_VARIABLE(remM_);
119 using urolls = unrolls::trans<Scalar>;
120
121 constexpr int64_t U3 = urolls::PacketSize * 3;
122 constexpr int64_t U2 = urolls::PacketSize * 2;
123 constexpr int64_t U1 = urolls::PacketSize * 1;
124
125 static_assert(unrollN == U1 || unrollN == U2 || unrollN == U3, "unrollN should be a multiple of PacketSize");
126 static_assert(unrollM == EIGEN_AVX_MAX_NUM_ROW, "unrollM should be equal to EIGEN_AVX_MAX_NUM_ROW");
127
128 urolls::template transpose<unrollN, 0>(zmm);
129 EIGEN_IF_CONSTEXPR(unrollN > U2) urolls::template transpose<unrollN, 2>(zmm);
130 EIGEN_IF_CONSTEXPR(unrollN > U1) urolls::template transpose<unrollN, 1>(zmm);
131
132 static_assert((remN && unrollN == U1) || !remN, "When handling N remainder set unrollN=U1");
133 EIGEN_IF_CONSTEXPR(!remN) {
134 urolls::template storeC<std::min(unrollN, U1), unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
135 EIGEN_IF_CONSTEXPR(unrollN > U1) {
136 constexpr int64_t unrollN_ = std::min(unrollN - U1, U1);
137 urolls::template storeC<unrollN_, unrollN, 1, remM>(C_arr + U1 * LDC, LDC, zmm, remM_);
138 }
139 EIGEN_IF_CONSTEXPR(unrollN > U2) {
140 constexpr int64_t unrollN_ = std::min(unrollN - U2, U1);
141 urolls::template storeC<unrollN_, unrollN, 2, remM>(C_arr + U2 * LDC, LDC, zmm, remM_);
142 }
143 }
144 else {
145 EIGEN_IF_CONSTEXPR((std::is_same<Scalar, float>::value)) {
146 // Note: without "if constexpr" this section of code will also be
147 // parsed by the compiler so each of the storeC will still be instantiated.
148 // We use enable_if in aux_storeC to set it to an empty function for
149 // these cases.
150 if (remN_ == 15)
151 urolls::template storeC<15, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
152 else if (remN_ == 14)
153 urolls::template storeC<14, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
154 else if (remN_ == 13)
155 urolls::template storeC<13, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
156 else if (remN_ == 12)

158 else if (remN_ == 11)
159 urolls::template storeC<11, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
160 else if (remN_ == 10)
161 urolls::template storeC<10, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
162 else if (remN_ == 9)
163 urolls::template storeC<9, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
164 else if (remN_ == 8)
165 urolls::template storeC<8, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
166 else if (remN_ == 7)
167 urolls::template storeC<7, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
168 else if (remN_ == 6)
169 urolls::template storeC<6, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
170 else if (remN_ == 5)
171 urolls::template storeC<5, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
172 else if (remN_ == 4)
173 urolls::template storeC<4, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
174 else if (remN_ == 3)
175 urolls::template storeC<3, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
176 else if (remN_ == 2)
177 urolls::template storeC<2, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
178 else if (remN_ == 1)
179 urolls::template storeC<1, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
180 }
181 else {
182 if (remN_ == 7)
183 urolls::template storeC<7, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
184 else if (remN_ == 6)
185 urolls::template storeC<6, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
186 else if (remN_ == 5)
187 urolls::template storeC<5, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
188 else if (remN_ == 4)
189 urolls::template storeC<4, unrollN, 0, remM>(C_arr, LDC, zmm, remM_);
190 else if (remN_ == 3)

192 else if (remN_ == 2)

196 }

214template <typename Scalar, bool isARowMajor, bool isCRowMajor, bool isAdd, bool handleKRem>
215void gemmKernel(Scalar *A_arr, Scalar *B_arr, Scalar *C_arr, int64_t M, int64_t N, int64_t K, int64_t LDA, int64_t LDB,
216 int64_t LDC) {
217 using urolls = unrolls::gemm<Scalar, isAdd>;
218 constexpr int64_t U3 = urolls::PacketSize * 3;
219 constexpr int64_t U2 = urolls::PacketSize * 2;
220 constexpr int64_t U1 = urolls::PacketSize * 1;
221 using vec = typename std::conditional<std::is_same<Scalar, float>::value, vecFullFloat, vecFullDouble>::type;
222 int64_t N_ = (N / U3) * U3;
223 int64_t M_ = (M / EIGEN_AVX_MAX_NUM_ROW) * EIGEN_AVX_MAX_NUM_ROW;
224 int64_t K_ = (K / EIGEN_AVX_MAX_K_UNROL) * EIGEN_AVX_MAX_K_UNROL;
225 int64_t j = 0;
226 for (; j < N_; j += U3) {
227 constexpr int64_t EIGEN_AVX_MAX_B_LOAD = EIGEN_AVX_B_LOAD_SETS * 3;
228 int64_t i = 0;
229 for (; i < M_; i += EIGEN_AVX_MAX_NUM_ROW) {
230 Scalar *A_t = &A_arr[idA<isARowMajor>(i, 0, LDA)], *B_t = &B_arr[0 * LDB + j];
231 PacketBlock<vec, EIGEN_ARCH_DEFAULT_NUMBER_OF_REGISTERS> zmm;
232 urolls::template setzero<3, EIGEN_AVX_MAX_NUM_ROW>(zmm);
233 for (int64_t k = 0; k < K_; k += EIGEN_AVX_MAX_K_UNROL) {
234 urolls::template microKernel<isARowMajor, 3, EIGEN_AVX_MAX_NUM_ROW, EIGEN_AVX_MAX_K_UNROL, EIGEN_AVX_MAX_B_LOAD,
235 EIGEN_AVX_MAX_A_BCAST>(B_t, A_t, LDB, LDA, zmm);
236 B_t += EIGEN_AVX_MAX_K_UNROL * LDB;
237 EIGEN_IF_CONSTEXPR(isARowMajor) A_t += EIGEN_AVX_MAX_K_UNROL;
238 else A_t += EIGEN_AVX_MAX_K_UNROL * LDA;
239 }
240 EIGEN_IF_CONSTEXPR(handleKRem) {
241 for (int64_t k = K_; k < K; k++) {
242 urolls::template microKernel<isARowMajor, 3, EIGEN_AVX_MAX_NUM_ROW, 1, EIGEN_AVX_B_LOAD_SETS * 3,
243 EIGEN_AVX_MAX_A_BCAST>(B_t, A_t, LDB, LDA, zmm);
244 B_t += LDB;
245 EIGEN_IF_CONSTEXPR(isARowMajor) A_t++;
246 else A_t += LDA;
247 }
248 }
249 EIGEN_IF_CONSTEXPR(isCRowMajor) {
250 urolls::template updateC<3, EIGEN_AVX_MAX_NUM_ROW>(&C_arr[i * LDC + j], LDC, zmm);
251 urolls::template storeC<3, EIGEN_AVX_MAX_NUM_ROW>(&C_arr[i * LDC + j], LDC, zmm);
252 }
253 else {
254 transStoreC<Scalar, vec, EIGEN_AVX_MAX_NUM_ROW, U3, false, false>(zmm, &C_arr[i + j * LDC], LDC);
255 }
256 }

725template <typename Scalar, typename vec, int64_t unrollM, bool isARowMajor, bool isFWDSolve, bool isUnitDiag>
726static EIGEN_ALWAYS_INLINE void triSolveKernel(Scalar *A_arr, Scalar *B_arr, int64_t K, int64_t LDA, int64_t LDB) {
727 static_assert(unrollM <= EIGEN_AVX_MAX_NUM_ROW, "unrollM should be equal to EIGEN_AVX_MAX_NUM_ROW");
728 using urolls = unrolls::trsm<Scalar>;
729 constexpr int64_t U3 = urolls::PacketSize * 3;
730 constexpr int64_t U2 = urolls::PacketSize * 2;
731 constexpr int64_t U1 = urolls::PacketSize * 1;
732
733 PacketBlock<vec, EIGEN_AVX_MAX_NUM_ACC> RHSInPacket;
734 PacketBlock<vec, EIGEN_AVX_MAX_NUM_ROW> AInPacket;
735
736 int64_t k = 0;
737 while (K - k >= U3) {
738 urolls::template loadRHS<isFWDSolve, unrollM, 3>(B_arr + k, LDB, RHSInPacket);
739 urolls::template triSolveMicroKernel<isARowMajor, isFWDSolve, isUnitDiag, unrollM, 3>(A_arr, LDA, RHSInPacket,
740 AInPacket);
741 urolls::template storeRHS<isFWDSolve, unrollM, 3>(B_arr + k, LDB, RHSInPacket);
742 k += U3;
743 }
744 if (K - k >= U2) {
745 urolls::template loadRHS<isFWDSolve, unrollM, 2>(B_arr + k, LDB, RHSInPacket);
746 urolls::template triSolveMicroKernel<isARowMajor, isFWDSolve, isUnitDiag, unrollM, 2>(A_arr, LDA, RHSInPacket,
747 AInPacket);
748 urolls::template storeRHS<isFWDSolve, unrollM, 2>(B_arr + k, LDB, RHSInPacket);
749 k += U2;
750 }
751 if (K - k >= U1) {
752 urolls::template loadRHS<isFWDSolve, unrollM, 1>(B_arr + k, LDB, RHSInPacket);
753 urolls::template triSolveMicroKernel<isARowMajor, isFWDSolve, isUnitDiag, unrollM, 1>(A_arr, LDA, RHSInPacket,
754 AInPacket);
755 urolls::template storeRHS<isFWDSolve, unrollM, 1>(B_arr + k, LDB, RHSInPacket);
756 k += U1;
757 }
758 if (K - k > 0) {
759 // Handle remaining number of RHS
760 urolls::template loadRHS<isFWDSolve, unrollM, 1, true>(B_arr + k, LDB, RHSInPacket, K - k);
761 urolls::template triSolveMicroKernel<isARowMajor, isFWDSolve, isUnitDiag, unrollM, 1>(A_arr, LDA, RHSInPacket,
762 AInPacket);
763 urolls::template storeRHS<isFWDSolve, unrollM, 1, true>(B_arr + k, LDB, RHSInPacket, K - k);
764 }
765}

776void triSolveKernelLxK(Scalar *A_arr, Scalar *B_arr, int64_t M, int64_t K, int64_t LDA, int64_t LDB) {
777 // Note: this assumes EIGEN_AVX_MAX_NUM_ROW = 8. Unrolls should be adjusted
778 // accordingly if EIGEN_AVX_MAX_NUM_ROW is smaller.

780 if (M == 8)
781 triSolveKernel<Scalar, vec, 8, isARowMajor, isFWDSolve, isUnitDiag>(A_arr, B_arr, K, LDA, LDB);
782 else if (M == 7)
783 triSolveKernel<Scalar, vec, 7, isARowMajor, isFWDSolve, isUnitDiag>(A_arr, B_arr, K, LDA, LDB);

785 triSolveKernel<Scalar, vec, 6, isARowMajor, isFWDSolve, isUnitDiag>(A_arr, B_arr, K, LDA, LDB);
786 else if (M == 5)
787 triSolveKernel<Scalar, vec, 5, isARowMajor, isFWDSolve, isUnitDiag>(A_arr, B_arr, K, LDA, LDB);
788 else if (M == 4)
789 triSolveKernel<Scalar, vec, 4, isARowMajor, isFWDSolve, isUnitDiag>(A_arr, B_arr, K, LDA, LDB);

871template <typename Scalar, bool isBRowMajor>
872constexpr std::pair<int64_t, int64_t> trsmBlocking(const int64_t limit) {
873 constexpr int64_t psize = packet_traits<Scalar>::size;
874 int64_t kB = 15 * psize;
875 int64_t numM = 8 * EIGEN_AVX_MAX_NUM_ROW;
876 // If B is rowmajor, no temp workspace needed, so use default blocking sizes.
877 if (isBRowMajor) return {kB, numM};
878
879 // Very simple heuristic, prefer keeping kB as large as possible to fully use vector registers.
880 for (int64_t k = kB; k > psize; k -= psize) {
881 for (int64_t m = numM; m > EIGEN_AVX_MAX_NUM_ROW; m -= EIGEN_AVX_MAX_NUM_ROW) {
882 if ((((k + psize - 1) / psize + 4) * psize) * m * sizeof(Scalar) < limit) {
883 return {k, m};
884 }
885 }
886 }
887 return {psize, EIGEN_AVX_MAX_NUM_ROW}; // Minimum blocking size required
888}

918template <typename Scalar, bool isARowMajor = true, bool isBRowMajor = true, bool isFWDSolve = true,
919 bool isUnitDiag = false>
920void triSolve(Scalar *A_arr, Scalar *B_arr, int64_t M, int64_t numRHS, int64_t LDA, int64_t LDB) {
921 constexpr int64_t psize = packet_traits<Scalar>::size;
935#if (EIGEN_USE_AVX512_TRSM_L_KERNELS) && defined(EIGEN_NO_MALLOC)
945 constexpr std::pair<int64_t, int64_t> blocking_ = trsmBlocking<Scalar, isBRowMajor>(EIGEN_STACK_ALLOCATION_LIMIT);
946 constexpr int64_t kB = blocking_.first;
947 constexpr int64_t numM = blocking_.second;
952 static_assert(!(((((kB + psize - 1) / psize + 4) * psize) * numM * sizeof(Scalar) >= EIGEN_STACK_ALLOCATION_LIMIT) &&
953 !isBRowMajor),
954 "Temp workspace required is too large.");
955#else
956 constexpr int64_t kB = (3 * psize) * 5; // 5*U3
957 constexpr int64_t numM = 8 * EIGEN_AVX_MAX_NUM_ROW;
958#endif
959
960 int64_t sizeBTemp = 0;
961 Scalar *B_temp = NULL;
962 EIGEN_IF_CONSTEXPR(!isBRowMajor) {
968 sizeBTemp = (((std::min(kB, numRHS) + psize - 1) / psize + 4) * psize) * numM;
969 }
970
971#if !defined(EIGEN_NO_MALLOC)
972 EIGEN_IF_CONSTEXPR(!isBRowMajor) B_temp = (Scalar *)handmade_aligned_malloc(sizeof(Scalar) * sizeBTemp, 64);
973#elif (EIGEN_USE_AVX512_TRSM_L_KERNELS) && defined(EIGEN_NO_MALLOC)
974 // Use alloca if malloc not allowed, requested temp workspace size should be less than EIGEN_STACK_ALLOCATION_LIMIT
975 ei_declare_aligned_stack_constructed_variable(Scalar, B_temp_alloca, sizeBTemp, 0);
976 B_temp = B_temp_alloca;
977#endif
978
979 for (int64_t k = 0; k < numRHS; k += kB) {
980 int64_t bK = numRHS - k > kB ? kB : numRHS - k;
981 int64_t M_ = (M / EIGEN_AVX_MAX_NUM_ROW) * EIGEN_AVX_MAX_NUM_ROW, gemmOff = 0;
982
983 // bK rounded up to next multiple of L=EIGEN_AVX_MAX_NUM_ROW. When B_temp is used, we solve for bkL RHS
984 // instead of bK RHS in triSolveKernelLxK.
985 int64_t bkL = ((bK + (EIGEN_AVX_MAX_NUM_ROW - 1)) / EIGEN_AVX_MAX_NUM_ROW) * EIGEN_AVX_MAX_NUM_ROW;
986 const int64_t numScalarPerCache = 64 / sizeof(Scalar);
987 // Leading dimension of B_temp, will be a multiple of the cache line size.
988 int64_t LDT = ((bkL + (numScalarPerCache - 1)) / numScalarPerCache) * numScalarPerCache;
989 int64_t offsetBTemp = 0;
990 for (int64_t i = 0; i < M_; i += EIGEN_AVX_MAX_NUM_ROW) {
991 EIGEN_IF_CONSTEXPR(!isBRowMajor) {
992 int64_t indA_i = isFWDSolve ? i : M - 1 - i;
993 int64_t indB_i = isFWDSolve ? i : M - (i + EIGEN_AVX_MAX_NUM_ROW);
994 int64_t offB_1 = isFWDSolve ? offsetBTemp : sizeBTemp - EIGEN_AVX_MAX_NUM_ROW * LDT - offsetBTemp;
995 int64_t offB_2 = isFWDSolve ? offsetBTemp : sizeBTemp - LDT - offsetBTemp;
996 // Copy values from B to B_temp.
997 copyBToRowMajor<Scalar, true, false>(B_arr + indB_i + k * LDB, LDB, bK, B_temp + offB_1, LDT);
998 // Triangular solve with a small block of A and long horizontal blocks of B (or B_temp if B col-major)
999 triSolveKernelLxK<Scalar, isARowMajor, isFWDSolve, isUnitDiag>(
1000 &A_arr[idA<isARowMajor>(indA_i, indA_i, LDA)], B_temp + offB_2, EIGEN_AVX_MAX_NUM_ROW, bkL, LDA, LDT);
1001 // Copy values from B_temp back to B. B_temp will be reused in gemm call below.
1002 copyBToRowMajor<Scalar, false, false>(B_arr + indB_i + k * LDB, LDB, bK, B_temp + offB_1, LDT);
1003
1004 offsetBTemp += EIGEN_AVX_MAX_NUM_ROW * LDT;
1005 }
1006 else {
1007 int64_t ind = isFWDSolve ? i : M - 1 - i;
1008 triSolveKernelLxK<Scalar, isARowMajor, isFWDSolve, isUnitDiag>(
1009 &A_arr[idA<isARowMajor>(ind, ind, LDA)], B_arr + k + ind * LDB, EIGEN_AVX_MAX_NUM_ROW, bK, LDA, LDB);
1010 }
1011 if (i + EIGEN_AVX_MAX_NUM_ROW < M_) {
1024 EIGEN_IF_CONSTEXPR(isBRowMajor) {
1025 int64_t indA_i = isFWDSolve ? i + EIGEN_AVX_MAX_NUM_ROW : M - (i + 2 * EIGEN_AVX_MAX_NUM_ROW);
1026 int64_t indA_j = isFWDSolve ? 0 : M - (i + EIGEN_AVX_MAX_NUM_ROW);
1027 int64_t indB_i = isFWDSolve ? 0 : M - (i + EIGEN_AVX_MAX_NUM_ROW);
1028 int64_t indB_i2 = isFWDSolve ? i + EIGEN_AVX_MAX_NUM_ROW : M - (i + 2 * EIGEN_AVX_MAX_NUM_ROW);
1029 gemmKernel<Scalar, isARowMajor, isBRowMajor, false, false>(
1030 &A_arr[idA<isARowMajor>(indA_i, indA_j, LDA)], B_arr + k + indB_i * LDB, B_arr + k + indB_i2 * LDB,
1031 EIGEN_AVX_MAX_NUM_ROW, bK, i + EIGEN_AVX_MAX_NUM_ROW, LDA, LDB, LDB);
1032 }
1033 else {
1034 if (offsetBTemp + EIGEN_AVX_MAX_NUM_ROW * LDT > sizeBTemp) {
1043 int64_t indA_i = isFWDSolve ? i + EIGEN_AVX_MAX_NUM_ROW : 0;
1044 int64_t indA_j = isFWDSolve ? gemmOff : M - (i + EIGEN_AVX_MAX_NUM_ROW);
1045 int64_t indB_i = isFWDSolve ? i + EIGEN_AVX_MAX_NUM_ROW : 0;
1046 int64_t offB_1 = isFWDSolve ? 0 : sizeBTemp - offsetBTemp;
1047 gemmKernel<Scalar, isARowMajor, isBRowMajor, false, false>(
1048 &A_arr[idA<isARowMajor>(indA_i, indA_j, LDA)], B_temp + offB_1, B_arr + indB_i + (k)*LDB,
1049 M - (i + EIGEN_AVX_MAX_NUM_ROW), bK, i + EIGEN_AVX_MAX_NUM_ROW - gemmOff, LDA, LDT, LDB);
1050 offsetBTemp = 0;
1051 gemmOff = i + EIGEN_AVX_MAX_NUM_ROW;
1052 } else {
1056 int64_t indA_i = isFWDSolve ? i + EIGEN_AVX_MAX_NUM_ROW : M - (i + 2 * EIGEN_AVX_MAX_NUM_ROW);
1057 int64_t indA_j = isFWDSolve ? gemmOff : M - (i + EIGEN_AVX_MAX_NUM_ROW);
1058 int64_t indB_i = isFWDSolve ? i + EIGEN_AVX_MAX_NUM_ROW : M - (i + 2 * EIGEN_AVX_MAX_NUM_ROW);
1059 int64_t offB_1 = isFWDSolve ? 0 : sizeBTemp - offsetBTemp;
1060 gemmKernel<Scalar, isARowMajor, isBRowMajor, false, false>(
1061 &A_arr[idA<isARowMajor>(indA_i, indA_j, LDA)], B_temp + offB_1, B_arr + indB_i + (k)*LDB,
1062 EIGEN_AVX_MAX_NUM_ROW, bK, i + EIGEN_AVX_MAX_NUM_ROW - gemmOff, LDA, LDT, LDB);
1063 }
1064 }
1065 }
1066 }
1067 // Handle M remainder..
1068 int64_t bM = M - M_;
1069 if (bM > 0) {
1070 if (M_ > 0) {
1071 EIGEN_IF_CONSTEXPR(isBRowMajor) {
1072 int64_t indA_i = isFWDSolve ? M_ : 0;
1073 int64_t indA_j = isFWDSolve ? 0 : bM;
1074 int64_t indB_i = isFWDSolve ? 0 : bM;
1075 int64_t indB_i2 = isFWDSolve ? M_ : 0;
1076 gemmKernel<Scalar, isARowMajor, isBRowMajor, false, false>(
1077 &A_arr[idA<isARowMajor>(indA_i, indA_j, LDA)], B_arr + k + indB_i * LDB, B_arr + k + indB_i2 * LDB, bM,
1078 bK, M_, LDA, LDB, LDB);
1079 }
1080 else {
1081 int64_t indA_i = isFWDSolve ? M_ : 0;
1082 int64_t indA_j = isFWDSolve ? gemmOff : bM;
1083 int64_t indB_i = isFWDSolve ? M_ : 0;
1084 int64_t offB_1 = isFWDSolve ? 0 : sizeBTemp - offsetBTemp;
1085 gemmKernel<Scalar, isARowMajor, isBRowMajor, false, false>(&A_arr[idA<isARowMajor>(indA_i, indA_j, LDA)],
1086 B_temp + offB_1, B_arr + indB_i + (k)*LDB, bM, bK,
1087 M_ - gemmOff, LDA, LDT, LDB);
1088 }
1089 }
1090 EIGEN_IF_CONSTEXPR(!isBRowMajor) {
1091 int64_t indA_i = isFWDSolve ? M_ : M - 1 - M_;
1092 int64_t indB_i = isFWDSolve ? M_ : 0;
1093 int64_t offB_1 = isFWDSolve ? 0 : (bM - 1) * bkL;
1094 copyBToRowMajor<Scalar, true, true>(B_arr + indB_i + k * LDB, LDB, bK, B_temp, bkL, bM);
1095 triSolveKernelLxK<Scalar, isARowMajor, isFWDSolve, isUnitDiag>(&A_arr[idA<isARowMajor>(indA_i, indA_i, LDA)],
1096 B_temp + offB_1, bM, bkL, LDA, bkL);
1097 copyBToRowMajor<Scalar, false, true>(B_arr + indB_i + k * LDB, LDB, bK, B_temp, bkL, bM);
1098 }
1099 else {
1100 int64_t ind = isFWDSolve ? M_ : M - 1 - M_;
1101 triSolveKernelLxK<Scalar, isARowMajor, isFWDSolve, isUnitDiag>(&A_arr[idA<isARowMajor>(ind, ind, LDA)],
1102 B_arr + k + ind * LDB, bM, bK, LDA, LDB);
1103 }
1104 }
1105 }
1106
1107#if !defined(EIGEN_NO_MALLOC)
1108 EIGEN_IF_CONSTEXPR(!isBRowMajor) handmade_aligned_free(B_temp);
1109#endif
